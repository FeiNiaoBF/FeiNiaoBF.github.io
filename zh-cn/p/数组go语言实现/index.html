func (arr *array) Bobble() {
 if arr.Len() <= 1 {
  return
 }
 for i := uint(0); i < arr.Len(); i++ {
  for j := uint(0); j < arr.Len()-i-1; j++ {
   if arr.data[j] > arr.data[j+1] {

 }
}

func (arr *array) BinarySearch(tar int) int {
 var (
  low  = 0
  high = int(arr.Len()) - 1
 )
 for low <= high {
  mid := low + (high-low)/2
  if arr.data[mid] > tar {
   high = mid - 1
  }else if arr.data[mid] < tar {
   low = mid + 1
  }else{
   return mid
  }
 }
 return -1
